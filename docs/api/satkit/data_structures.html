 94class Recording:
 95    """
 96    A Recording contains 0-n synchronised Modalities.
 97
 98    The recording also contains the non-modality 
 99    specific metadata (participant, speech content, etc) 
100    as a dictionary, as well as the textgrid for the whole recording.
101
102    In inheriting classes call self._read_textgrid() after calling
103    super.__init__() (with correct arguments) and doing any updates
104    to self.meta['textgrid'] that are necessary.
105    """
106
107    def __init__(self,
108                 meta_data: RecordingMetaData,
109                 excluded: bool = False,
110                 textgrid_path: Union[str, Path] = "") -> None:
111        """
112        Construct a mainly empty recording without modalities.
113
114        Modalities and annotations get added after constructions with their own
115        add_[modality or annotation] functions.
116
117        Parameters
118        ----------
119        meta_data : RecordingMetaData
120            Some of the contents of the meta data are avaible as properties.
121        excluded : bool, optional
122            _description_, by default False
123        textgrid_path : Union[str, Path], optional
124            _description_, by default ""
125        """
126        self.excluded = excluded
127
128        self.meta_data = meta_data
129
130        if textgrid_path:
131            self._textgrid_path = textgrid_path
132        else:
133            self._textgrid_path = meta_data.path.joinpath(
134                meta_data.basename + ".TextGrid")
135        self.textgrid = self._read_textgrid()
136        self.satgrid = SatGrid(self.textgrid)
137
138        self.modalities = {}
139        self.annotations = {}
140

143        return self.meta_data.path

146    def path(self, path: Path) -> None:
147        self.meta_data.path = path
148

150    def basename(self) -> str:
151        return self.meta_data.basename

153    @basename.setter
154    def basename(self, basename: str) -> None:
155        self.meta_data.basename = basename

157    def _read_textgrid(self) -> textgrids.TextGrid:
158        """
159        Read the textgrid specified in self.meta.
160
161        If file does not exist or reading fails, recovery is attempted 
162        by logging an error and creating an empty textgrid for this 
163        recording.
164        """
165        textgrid = None
166        if self._textgrid_path.is_file():
167            try:
168                textgrid = textgrids.TextGrid(self._textgrid_path)
169                _datastructures_logger.info("Read textgrid in %s.",
170                                            self._textgrid_path)
171            except Exception as e:
172                _datastructures_logger.critical(
173                    "Could not read textgrid in %s.", self._textgrid_path)
174                _datastructures_logger.critical("Failed with: %s.", str(e))
175                _datastructures_logger.critical("Creating an empty textgrid "
176                                                + "instead.")
177                textgrid = textgrids.TextGrid()
178        else:
179            notice = 'Note: ' + str(self._textgrid_path) + " did not exist."
180            _datastructures_logger.warning(notice)
181            _datastructures_logger.warning("Creating an empty textgrid "
182                                           + "instead.")
183            textgrid = textgrids.TextGrid()
184        return textgrid
185
186    def exclude(self) -> None:
187        """
188        Set self.excluded to True with a method.
189
190        This method exists to facilitate list comprehensions being used
191        for excluding recordings e.g. 
192        [recording.exclude() for recording in recordings if in some_list].
193        """
194        self.excluded = True

196    def write_textgrid(self, filepath: Path = None) -> None:
197        """
198        Save this recording's textgrid to file.
199
200        Keyword argument:
201        filepath -- string specifying the path and name of the 
202            file to be written. If filepath is not specified, this 
203            method will try to overwrite the textgrid speficied in 
204            self.meta.
205
206            If filepath is specified, subsequent calls to this 
207            function will write into the new path rather than 
208            the original one.
209        """
210        try:
211            if filepath:
212                self.textgridpath = filepath
213                self.textgrid.write(filepath)
214            else:
215                self.textgrid.write(self.textgridpath)
216        except Exception as e:
217            _datastructures_logger.critical("Could not write textgrid to "
218                                            + str(self.textgridpath) + ".")
219            _datastructures_logger.critical("TextGrid save failed with error: "
220                                            + str(e))

223    def add_modality(
224            self, modality: 'Modality', replace: bool = False) -> None:
225        """
226        This method adds a new Modality object to the Recording.
227
228        Replacing a modality has to be specified otherwise if a
229        Modality with the same name already exists in this Recording
230        and the replace argument is not True, an Error is raised. 
231
232        Arguments:
233        modality -- object of type Modality to be added to 
234            this Recording.
235
236        Keyword arguments:
237        replace -- a boolean indicating if an existing Modality should
238            be replaced.
239        """
240        name = modality.name
241
242        if name in self.modalities.keys() and not replace:
243            raise ModalityError(
244                "A modality named " + name +
245                " already exists and replace flag was False.")
246        elif replace:
247            self.modalities[name] = modality
248            _datastructures_logger.debug("Replaced modality " + name + ".")
249        else:
250            self.modalities[name] = modality
251            _datastructures_logger.debug("Added new modality " + name + ".")
252
253    def __str__(self) -> str:
254        """
255        Bare minimum string representation of the Recording.
256
257        Returns
258        -------
259        string
260            'Recording [basename]'
261        """
262        return f"Recording {self.basename}"

270    def __init__(self,
271                 recording: Recording,
272                 parsed_data: Optional[ModalityData] = None,
273                 meta_data: Optional[ModalityMetaData] = None,
274                 data_path: Optional[Path] = None,
275                 meta_path: Optional[Path] = None,
276                 load_path: Optional[Path] = None,
277                 time_offset: Optional[float] = None
278                 ) -> None:
279        """
280        Modality constructor.
281
282        Positional arguments:
283        recording -- the containing Recording.
284
285        Keyword arguments:
286        data_path -- path of the data file
287        load_path -- path of data when saved by SATKIT - both data and metadata
288        parent -- the Modality this one was derived from. None means this 
289            is an underived data Modality.
290        parsed_data -- ModalityData object containing waveform, sampling rate,
291            and either timevector and/or time_offset. 
292        parsed_data -- a ModalityData object containing parsed data 
293            that's been either read from file, loaded from file 
294            (previously saved by SATKIT), or calculated from another modality.
295            Providing a timevector 
296            overrides any time_offset value given, but in absence of a 
297            timevector the time_offset will be applied on reading the data 
298            from file. 
299        time_offset -- offset of this modality in relation to the Recordings
300            baseline - usually the audio track. This will be ignored if 
301            parsed_data exists and effectively overridden by 
302            parsed_data.timevector.
303        """
304        self.recording = recording
305        self.data_path = data_path
306        self._meta_path = meta_path  # self.meta_path is a property
307        self.load_path = load_path
308
309        self.meta_data = meta_data
310
311        if parsed_data:
312            self._data = parsed_data.data
313            self._sampling_rate = parsed_data.sampling_rate
314            self._timevector = parsed_data.timevector
315        else:
316            self._data = None
317            self._sampling_rate = None
318            self._timevector = None
319
320        if self._timevector is None:
321            self._time_offset = time_offset
322        else:
323            self._time_offset = self._timevector[0]
324
325        # This is a property which when set to True will also set
326        # parent.excluded to True.
327        self.excluded = False

329    def _get_data(self) -> ModalityData:
330        # TODO: Provide a way to force the data to be derived.
331        # this would be used when parent modality has updated in some way
332        if self.data_path:
333            return self._read_data()
334        elif self.load_path:
335            return self._load_data()
336        elif self.meta_data.parent_name:
337            return self._derive_data()
338        else:
339            raise MissingDataError(
340                "Asked to get data but have no path and no parent Modality.\n"
341                + "Don't know how to solve this.")

346    def _derive_data(self) -> ModalityData:
347        """
348        Derive data from another modality -- to be overridden by inheriting classes.
349
350        This method should be implemented by subclasses by calling 
351        a suitable deriving function to provide on-the-fly loading of data.
352
353        This method is intended to rely on self.parent to provide the data
354        to be processed.
355        """
356        raise NotImplementedError(
357            "This method should be overridden by inheriting classes.")

359    def _load_data(self) -> ModalityData:
360        """
361        Load data from a saved file -- to be overridden by inheriting classes.
362
363        This method should be implemented by subclasses by calling 
364        a suitable loading function to provide on-the-fly loading of data.
365
366        This method is intended to rely on self.load_path to know what to read.
367        """
368        raise NotImplementedError(
369            "This method should be overridden by inheriting classes.")

371    def _read_data(self) -> ModalityData:
372        """
373        Load data from file -- to be overridden by inheriting classes.
374
375        This method should be implemented by subclasses by calling 
376        a suitable reading function to provide on-the-fly loading of data.
377
378        This method is intended to rely on self.data_path to know what to read.
379        """
380        raise NotImplementedError(
381            "This method should be overridden by inheriting classes.")
382
383    def _set_data(self, data: ModalityData):
384        """Method used to set data from _get_data, _load_data, and _derive_data."""
385        self._data = data.data
386        self._timevector = data.timevector
387        self._sampling_rate = data.sampling_rate
388
389    @property
390    def name(self) -> str:
391        """
392        Identity and possible parent data class.
393
394        This will be just the class name if this is a data Modality instance.
395        For derived Modalities the name will be of the form
396        '[own class name] on [data modality class name]'.
397
398        Subclasses may override this behaviour to, for example, include
399        the metric used to generate the instance in the name.
400        """
401        name_string = self.__class__.__name__
402        if self.meta_data and self.meta_data.parent_name:
403            name_string = name_string + " on " + self.meta_data.parent_name
404        return name_string
405
406    @property
407    def data(self) -> np.ndarray:
408        """
409        Abstract property: a NumPy array. 
410
411        The data refers to the actual data this modality represents
412        and for DerivedModality it is the result of running the 
413        modality's algorithm on the original data.
414
415        The dimensions of the array are in the 
416        order of [time, others]
417
418        If this modality is not preloaded, accessing this property will
419        cause data to be loaded on the fly _and_ saved in memory. To 
420        release the memory, assign None to this Modality's data.
421        """
422        if self._data is None:
423            _datastructures_logger.debug(
424                "In Modality data getter. self._data was None.")
425            self._set_data(self._get_data())
426        return self._data
427
428    @data.setter
429    def data(self, data: np.ndarray) -> None:
430        """
431        Data setter method.
432
433        Arguments:
434        data - either None or a numpy.ndarray with same dtype, size, 
435            and shape as self.data.
436
437        Assigning anything but None or a numpy ndarray with matching
438        dtype, size, and shape will raise a OverWriteError.
439
440        If shape of the data were to change then also shape of the timevector should
441        potentially change. Unlikely that we'd try to deal that in any other way but
442        to create a new Modality or even Recording.
443        """
444        self._data_setter(data)
445
446    def _data_setter(self, data: np.ndarray) -> None:
447        """Set the data property from this class and subclasses."""
448        if self.data is not None and data is not None:
449            if (data.dtype == self._data.dtype and data.size == self._data.size and
450                    data.shape == self._data.shape):
451                self._data = data
452            else:
453                raise OverWriteError(
454                    "Trying to write over raw ultrasound data with " +
455                    "a numpy array that has non-matching dtype, size, or shape.\n" +
456                    " data.shape = " + str(data.shape) + "\n" +
457                    " self.data.shape = " + str(self._data.shape) + ".")
458        else:
459            self._data = data

475    @property
476    def time_offset(self):
477        """
478        The time offset of this modality.
479
480        Assigning a value to this property is implemented so 
481        that self._timevector[0] stays equal to self._timeOffset. 
482
483        If shape of the timevector were to change then also shape of the data should
484        change. Unlikely that we'd try to deal that in any other way but
485        to create a new Modality or even Recording.
486        """
487        if not self._time_offset:
488            self._set_data(self._get_data())
489        return self._time_offset
490
491    @time_offset.setter
492    def time_offset(self, timeOffset):
493        self._time_offset = timeOffset
494        if self.timevector:
495            self._timevector = (self.timevector +
496                                (timeOffset - self.timevector[0]))
497
498    @property
499    def timevector(self):
500        """
501        The timevector corresponding to self.data as a NumPy array. 
502
503        If the data has not been previously loaded, accessing this 
504        property will cause data to be loaded on the fly _and_ saved 
505        in memory. To release the memory, assign None to this 
506        Modality's data. If the data has been previously 
507        loaded and after that released, the timevector still persists and  
508        accessing it does not trigger a new loading operation.
509
510        Assigning a value to this property is implemented so 
511        that self._timevector[0] stays equal to self._timeOffset. 
512        """
513        if self._timevector is None:
514            self._set_data(self._get_data())
515        return self._timevector
516
517    @timevector.setter
518    def timevector(self, timevector):
519        if self._timevector is None:
520            raise OverWriteError(
521                "Trying to overwrite the time vector when it has not yet been initialised."
522            )
523        elif timevector is None:
524            raise NotImplementedError(
525                "Trying to set timevector to None.\n" +
526                "Freeing timevector memory is currently not implemented."
527            )
528        else:
529            if (timevector.dtype == self._timevector.dtype and
530                timevector.size == self._timevector.size and
531                    timevector.shape == self._timevector.shape):
532                self._timevector = timevector
533                self.time_offset = timevector[0]
534            else:
535                raise OverWriteError(
536                    "Trying to write over raw ultrasound data with " +
537                    "a numpy array that has non-matching dtype, size, or shape.\n" +
538                    " timevector.shape = " + str(timevector.shape) + "\n" +
539                    " self.timevector.shape = " + str(self._timevector.shape) + ".")

589def satkit_suffix(satkit_type: Union[Recording, RecordingSession, Modality]):
590    # TODO 1.1: This is one possibility for not having hardcoded file suffixes.
591    # Another is to let all the classes take care of it themselves and make it into
592    # a Protocol (Python version of an interface).
593    suffix = SatkitSuffix.META
594    if satkit_type == Recording:
595        suffix = '.Recording' + suffix
596    elif satkit_type == RecordingSession:
597        suffix = '.RecordingSession' + suffix
598    elif satkit_type == Modality:
599        suffix = ''
600    return suffix

Positional arguments: recording -- the containing Recording.

Keyword arguments: data_path -- path of the data file load_path -- path of data when saved by SATKIT - both data and metadata parent -- the Modality this one was derived from. None means this is an underived data Modality. parsed_data -- ModalityData object containing waveform, sampling rate, and either timevector and/or time_offset. parsed_data -- a ModalityData object containing parsed data that's been either read from file, loaded from file (previously saved by SATKIT), or calculated from another modality. Providing a timevector overrides any time_offset value given, but in absence of a timevector the time_offset will be applied on reading the data from file. time_offset -- offset of this modality in relation to the Recordings baseline - usually the audio track. This will be ignored if parsed_data exists and effectively overridden by parsed_data.timevector.
