 50class DataObject(abc.ABC):
 51    """
 52    Abstract base class for SATKIT data objects.
 53
 54    Almost no class should directly inherit from this class. Exceptions are
 55    DataAggregator and DataContainer. The latter is the abstract baseclass for
 56    Modality and Statistic and the former for all data base classes: Recording,
 57    Session, DataSet and any others that contain either DataContainers and/or
 58    DataAggregators.
 59    """
 60
 61    def __init__(self,
 62                 metadata: SatkitBaseModel,
 63                 owner: DataAggregator | None = None,
 64                 file_info: FileInformation | None = None,
 65                 ) -> None:
 66        # The super().__init__() call below is needed to make sure that
 67        # inheriting classes which also inherit from UserDict and UserList are
 68        # initialised properly.
 69        super().__init__()
 70
 71        self._metadata = metadata
 72        self.owner = owner
 73        self._file_info = file_info
 74
 75    def __getstate__(self) -> dict:
 76        """
 77        Return this DataContainer's pickle compatible state.
 78
 79        To achieve pickle compatibility, subclasses should take care to delete
 80        any cyclical references, like is done with `self.owner` here.
 81
 82        NOTE! This also requires owner to be reset after unpickling by the
 83        owners, because the unpickled class can not know who owns it.
 84
 85        Returns
 86        -------
 87        dict
 88            The state without cyclical references.
 89        """
 90        state = self.__dict__.copy()
 91        del state['owner']
 92        return state
 93
 94    @property
 95    @abc.abstractmethod
 96    def name(self) -> str:
 97        """
 98        Name of this instance.
 99
100        In most cases name is supposed to be implemented with the following
101        idiom:
102        ```python
103        return NAME_OF_THIS_CLASS.generate_name(self.metadata)
104        ```
105        For example, for PD:
106        ```python
107        return PD.generate_name(self.metadata)
108        ```
109
110        Returns
111        -------
112        str
113            Name of this instance.
114        """

385class DataAggregator(DataObject):
386    """
387    Abstract baseclass for Recording, Session, and DataSet. 
388
389    This class collects behaviors that are shared by the data base classes i.e.
390    classes which collect DataContainers and/or DataAggregators.
391    """
392
393    def __init__(self,
394                 name: str,
395                 metadata: SatkitBaseModel,
396                 owner: DataObject | None = None,
397                 file_info: FileInformation | None = None,
398                 statistics: dict[str, 'Statistic'] | None = None
399                 ) -> None:
400        super().__init__(
401            owner=owner, metadata=metadata, file_info=file_info)
402
403        self._name = name
404        self.statistics = {}
405        if statistics:
406            self.statistics.update(statistics)
407
408    @property
409    def name(self) -> str:
410        """
411        Name of this instance.
412
413        DataAggregators get their names mainly from the file system. DataSets
414        are named after the root directory name, Sessions for the session
415        directories and Trials for the trial file names. 
416
417        Returns
418        -------
419        str
420            The name as string.
421        """
422        return self._name
423
424    def add_statistic(
425            self, statistic: 'Statistic', replace: bool = False) -> None:
426        """
427        Add a Statistic to this Session.
428
429        Parameters
430        ----------
431        statistic : Statistic
432            Statistic to be added.
433        replace : bool, optional
434            Should we replace any existing Statistic by the same name, by
435            default False
436
437        Raises
438        ------
439        OverwriteError
440            In case replace was False and there exists already a Statistic with
441            the same name in this Session.
442        """
443        name = statistic.name
444
445        if name in self.statistics and not replace:
446            raise OverwriteError(
447                "A Statistic named '" + name +
448                "' already exists and replace flag was False.")
449
450        if replace:
451            self.statistics[name] = statistic
452            _datastructures_logger.debug("Replaced statistic %s.", name)
453        else:
454            self.statistics[name] = statistic
455            _datastructures_logger.debug("Added new statistic %s.", name)

458class DataContainer(DataObject):
459    """
460    Abstract baseclass for Modality and Statistic. 
461
462    This class collects behaviors shared by the classes that contain data:
463    Modalities contain time varying data and Statistics contain time
464    independent data.
465    """

467    @abc.abstractmethod
468    def generate_name(cls, params: SatkitBaseModel) -> str:
469        """Abstract version of generating a RecordingMetric name."""
470
471    def __init__(self,
472                 metadata: SatkitBaseModel,
473                 owner: DataObject | None = None,
474                 file_info: FileInformation | None = None,
475                 ) -> None:
476        super().__init__(
477            owner=owner, metadata=metadata, file_info=file_info)
478
479    @property
480    def name(self) -> str:
481        """
482        Name of this instance.
483
484        In most cases name is supposed to be overridden with the following
485        idiom:
486        ```python
487        return NAME_OF_THIS_CLASS.generate_name(self.metadata)
488        ```
489        For example, for PD:
490        ```python
491        return PD.generate_name(self.metadata)
492        ```
493
494        Returns
495        -------
496        str

498        """
499        return self.__class__.generate_name(self._metadata)

522class Statistic(DataContainer):
523    """
524    Abstract baseclass for statistics generated from members of a container. 
525
526    Specifically Statistics are time independent data while Modalities are
527    time-dependent data.
528    """
529    _data: np.ndarray
530
531    @classmethod
532    @abc.abstractmethod
533    def generate_name(cls, params: StatisticMetaData) -> str:
534        """Abstract version of generating a Statistic name."""
535
536    def __init__(
537            self,
538            metadata: SatkitBaseModel,
539            owner: DataAggregator | None = None,
540            file_info: FileInformation | None = None,
541            parsed_data: np.ndarray | None = None,
542    ) -> None:
543        """
544        Build a Statistic.       
545
546        Parameters
547        ----------
548        metadata : SatkitBaseModel
549            Parameters used in calculating this Statistic.
550        owner : DataAggregator
551            The owner of this Statistic. Usually this will be the object whose
552            contents this Statistic was calculated on. By default, None, to
553            facilitate mass generation and setting the owner after wards.
554        file_info : FileInformation
555            The SATKIT load path and names for this Statistic. Recorded path and
556            names should usually be empty. Defaults to None, when the Statistic
557            hasn't been saved yet.
558        parsed_data : Optional[np.ndarray], optional
559            the actual statistic, by default None
560        """
561        super().__init__(
562            owner=owner, metadata=metadata, file_info=file_info)
563        self._data = parsed_data
