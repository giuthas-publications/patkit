 62def _load_data_container_data_and_meta(
 63    path: Path,
 64    data_container_schema: DataContainerListingLoadSchema,
 65) -> tuple[FileInformation, DataContainerLoadSchema, Any]:
 66    """
 67    Load file information, meta data, and actual data for a DataContainer.
 68
 69    Only a helper function, not for general use.
 70    """
 71    file_info = FileInformation(
 72        patkit_path=Path(""),
 73        patkit_data_file=data_container_schema.data_name,
 74        patkit_meta_file=data_container_schema.meta_name,
 75    )
 76    meta_path = path / data_container_schema.meta_name
 77    data_path = path / data_container_schema.data_name
 78
 79    raw_input = nestedtext.load(meta_path)
 80    meta = DataContainerLoadSchema.model_validate(raw_input)
 81
 82    saved_data = np.load(data_path)
 83
 84    return file_info, meta, saved_data

 87def load_derived_modality(
 88        recording: Recording,
 89        path: Path,
 90        modality_schema: DataContainerListingLoadSchema
 91) -> None:
 92    """
 93    Load a saved derived Modality meta and data, and add them to the Recording.
 94
 95    Parameters
 96    ----------
 97    recording : Recording
 98        The Recording the Modality will be added to.
 99    path : Path
100        This is the path to the save files.
101    modality_schema : DataContainerListingLoadSchema
102        This contains the name of the meta and data files.
103    """
104    if not modality_schema.meta_name:
105        _logger.info(
106            "Looks like %s doesn't have a metafile for one of the Modalities.",
107            modality_schema.data_name)
108        _logger.info(
109            "Assuming the Modality to be batch loaded, so skipping.")
110        return
111
112    file_info, meta, saved_data = _load_data_container_data_and_meta(
113        path, modality_schema)
114
115    modality_data = ModalityData(
116        saved_data['data'], sampling_rate=saved_data['sampling_rate'],
117        timevector=saved_data['timevector'])
118
119    modality_constructor, parameter_schema = metrics[meta.object_type]
120    for key in meta.parameters:
121        if meta.parameters[key] == 'None':
122            meta.parameters[key] = None
123    parameters = parameter_schema(**meta.parameters)
124    modality = modality_constructor(
125        container=recording,
126        file_info=file_info,
127        parsed_data=modality_data,
128        metadata=parameters)
129
130    recording.add_modality(modality=modality)

198def load_recording(
199    patkit_path: Path,
200    recorded_path: Path,
201    container: Session,
202) -> Recording:
203    """
204    Load a recording from given Path.
205
206    Parameters
207    ----------
208    patkit_path : Path
209        Path to Recording's saved metadata file.
210    recorded_path : Path
211        Path to Recording's recorded data.
212
213    Returns
214    -------
215    Recording
216        A Recording object with most of its modalities loaded. Modalities like
217        Splines that maybe stored in one file for several recordings aren't yet
218        loaded at this point.
219
220    Raises
221    ------
222    NotImplementedError
223        If there is no previously saved metadata for the recording. This maybe
224        handled by a future version of patkit, if it should prove necessary.
225    """
226    # decide which loader we will be using based on either filepath.patkit_meta
227    # or config[''] in that order and document this behaviour. this way if the
228    # data has previously been loaded patkit can decide itself what to do with
229    # it and there is an easy place where to add processing
230    # session/participant/whatever specific config. could also add guessing
231    # based on what is present as the final fall back or as the option tried if
232    # no meta and config has the wrong guess.
233
234    meta_path = patkit_path.with_suffix(PatkitSuffix.META)
235    if meta_path.is_file():
236        # this is a list of Modalities, each with a data path and meta path
237        recording_meta = read_recording_meta(patkit_path)
238    else:
239        # TODO: need to hand to the right kind of importer here.
240        raise NotImplementedError(
241            "Can't yet jump to a previously unloaded recording here.")
242
243    file_info = FileInformation(
244        recorded_path=recorded_path,
245        patkit_path=meta_path.parent,
246        patkit_meta_file=meta_path.name)
247    recording = Recording(
248        metadata=recording_meta.parameters,
249        file_info=file_info,
250        container=container
251    )
252
253    for modality in recording_meta.modalities:
254        if modality in modality_adders:
255            adder = modality_adders[modality]
256            path = recorded_path/recording_meta.modalities[modality].data_name
257            adder(recording, path=path)
258        else:
259            load_derived_modality(
260                recording=recording,
261                path=patkit_path.parent,
262                modality_schema=recording_meta.modalities[modality])
263    for statistic in recording_meta.statistics:
264        load_statistic(
265            recording,
266            path=patkit_path.parent,
267            statistic_schema=recording_meta.statistics[statistic])
268
269    return recording

272def load_recordings(
273        patkit_path: Path,
274        recorded_path: Path,
275        container: Session,
276        recording_metafiles: list[str] | None = None
277) -> list[Recording]:
278    """
279    Load (specified) Recordings from directory.
280
281    Parameters
282    ----------
283    directory : Path
284        Path to the directory.
285    recording_metafiles : Optional[list[str]]
286        Names of the Recording metafiles. If omitted, all Recordings in the
287        directory will be loaded.
288
289    Returns
290    -------
291    list[Recording]
292        List of the loaded Recordings.
293    """
294    if not recording_metafiles:
295        recording_metafiles = patkit_path.glob(
296            "*.Recording" + str(PatkitSuffix.META))
297
298    recordings = [
299        load_recording(
300            patkit_path=patkit_path / name,
301            recorded_path=recorded_path,
302            container=container,
303        )
304        for name in recording_metafiles]
305
306    add_splines(recordings, patkit_path)
307
308    return recordings

311def load_recording_session(
312        directory: Path | str,
313        session_config_path: Path | None = None
314) -> Session:
315    """
316    Load a recording session from a directory.
317
318    Parameters
319    ----------
320    directory: Path
321        Root directory of the data.
322    session_config_path : Path | None
323        Path to the session configuration file. By default, None.
324
325    Returns
326    -------
327    Session
328        The loaded Session object.
329    """
330    # TODO: data_loader and this function should have clearer split of
331    # responsibilities
332    if isinstance(directory, str):
333        directory = Path(directory)
334
335    if session_config_path is None:
336        session_config_path = directory / PatkitConfigFile.SESSION
337
338    # filename = f"{directory.parts[-1]}{'.Session'}{PatkitSuffix.META}"
339    # filepath = directory / filename
340
341    filename = f"Session{PatkitSuffix.META}"
342    filepath = directory / filename
343
344    raw_input = nestedtext.load(filepath)
345    meta = SessionLoadSchema.model_validate(raw_input)
346
347    # if session_config_path.is_file():
348    #     session_config = load_session_config(
349    #         directory, session_config_path)
350    # else:
351    session_config = SessionConfig(
352        data_source_name=meta.parameters.datasource_name,
353        path_structure=meta.parameters.path_structure
354    )
355
356    file_info = FileInformation(
357        patkit_meta_file=filename,
358        patkit_path=directory,
359        recorded_path=meta.parameters.recorded_path,
360        recorded_meta_file=session_config_path.name
361    )
362    session = Session(
363        name=meta.name,
364        config=session_config,
365        file_info=file_info
366    )
367
368    recordings = load_recordings(
369        patkit_path=meta.parameters.patkit_path,
370        recorded_path=meta.parameters.recorded_path,
371        container=session,
372        recording_metafiles=meta.recordings,
373    )
374
375    session.extend(recordings)
376
377    return session
