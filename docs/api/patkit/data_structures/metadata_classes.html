 54@dataclass
 55class FileInformation:
 56    """
 57    File and Path information for patkit DataObjects.
 58
 59    recorded_data_file: str | None = None
 60        Name of the file containing the raw recorded data.
 61    recorded_meta_file: str | None = None
 62        Name of the file containing the meta data of the recording.
 63    recorded_path : Path | None = None
 64        Path to the recorded data of this DataObject - if there is original
 65        recorded data associated with this instance/type. Defaults to None
 66    patkit_data_file : str | None
 67        Name of the patkit data file, if it exists. Defaults to None.
 68    patkit_meta_file : str | None
 69        Name of the patkit meta file, if it exists. Defaults to None.
 70    patkit_path : Path | None
 71        Path to the saved patkit data, if it exists. Defaults to None.
 72    """
 73    recorded_data_file: str | None = None
 74    recorded_meta_file: str | None = None
 75    recorded_path: Path | None = None
 76    patkit_data_file: str | None = None
 77    patkit_meta_file: str | None = None
 78    patkit_path: Path | None = None
 79
 80    @property
 81    def basename(self) -> str:
 82        """
 83        Name of either the recorded or patkit data.
 84
 85        Recorded takes precedence if for some reason both exist.
 86
 87        Returns
 88        -------
 89        str
 90            Name without suffix
 91        """
 92        return self.basepath.name
 93
 94    @property
 95    def basepath(self) -> Path:
 96        """
 97        Path of the recorded or patkit data without the suffix.
 98
 99        This is a concatenation of the local path part (does not contain e.g. a
100        Modality's parent's path) and the name with the suffix(es) dropped.
101
102        Returns
103        -------
104        Path
105            Data Path without suffix.
106        """
107        basepath = None
108        if self.recorded_path:
109            if self.recorded_data_file:
110                basepath = self.recorded_path/self.recorded_data_file
111            elif self.recorded_meta_file:
112                basepath = self.recorded_path/self.recorded_meta_file
113        if basepath is None:
114            if self.patkit_meta_file is not None:
115                basepath = self.patkit_path/self.patkit_meta_file
116        return stem_path(basepath)

122    Data passed from Modality generation into Modality.

130    r-phi-confidence, spline points] for data in polar coordinates.
131    """

133    sampling_rate: float
134    timevector: np.ndarray

148class PointAnnotations:
149    """
150    Time point annotations for a Modality.
151
152    For each modality there should be only one of these for each kind of
153    annotation type.
154
155    annotation_type : AnnotationType
156        unique identifier for the annotation type
157    indeces : np.ndarray
158        indeces of the annotation points. `modality_data.data[indeces[i]]` and
159        `modality_data.timevector[indeces[i]]` correspond to the annotation at
160        `i`.
161    times : np.ndarray
162        timestamps of the annotation points
163    generating_parameters : dict
164        the function call arguments and other parameters used in generating
165        these annotations.
166    properties : dict
167        a dictionary containing arrays of each of the annotation properties
168        expected for this annotation type.
169    """
170    annotation_type: AnnotationType
171    indeces: np.ndarray
172    times: np.ndarray
173    generating_parameters: PointAnnotationParams
174    properties: dict
175
176    def add_annotation(
177            self, index: int, time: float, properties: dict) -> None:
178        """
179        This method has not been implemented yet.
180
181        Index and time should be mutually exclusive.
182
183        Parameters
184        ----------
185        index : int
186            index at which the annotation is to be added
187        time : float
188            time at which the annotation is to be added
189        properties : dict
190            the annotation properties that will be added to the arrays in this
191            PointAnnotations' properties dict.
192
193        Raises
194        ------
195        NotImplementedError
196            This method has not been implemented yet.
197        """
198        raise NotImplementedError(
199            "Adding annotations to "
200            "PointAnnotations hasn't been implemented yet.")
201
202    def apply_lower_time_limit(self, time_min: float) -> None:
203        """
204        Apply a lower time limit to the annotations.
205
206        This removes the annotation points before the given time limit.
207
208        Parameters
209        ----------
210        time_min : float
211            The time limit.
212        """
213        selected = np.nonzero(self.times >= time_min)
214        self.indeces = self.indeces[selected]
215        self.times = self.times[selected]
216        limit = selected[0]
217
218        for key in self.properties:
219            if is_sequence_form(self.properties[key]):
220                self.properties[key] = self.properties[key][limit:]
221            elif isinstance(self.properties[key], np.ndarray):
222                self.properties[key] = self.properties[key][selected]
223
224    def apply_upper_time_limit(self, time_max: float) -> None:
225        """
226        Apply an upper time limit to the annotations.
227
228        This removes the annotation points after the given time limit.
229
230        Parameters
231        ----------
232        time_max : float
233            The time limit.
234        """
235        selected = np.nonzero(self.times <= time_max)
236        self.indeces = self.indeces[selected]
237        self.times = self.times[selected]
238        limit = selected[-1]
239
240        for key in self.properties:
241            if is_sequence_form(self.properties[key]):
242                self.properties[key] = self.properties[key][:limit]
243            elif isinstance(self.properties[key], np.ndarray):
244                self.properties[key] = self.properties[key][selected]
