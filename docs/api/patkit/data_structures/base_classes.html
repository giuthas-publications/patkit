 50class AbstractDataObject(abc.ABC):
 51    """
 52    Abstract base class for patkit data objects.
 53
 54    Almost no class should directly inherit from this class. Exceptions are
 55    AbstractDataContainer and AbstractData. The latter is the abstract baseclass for
 56    Modality and Statistic and the former for all data base classes: Recording,
 57    Session, DataSet and any others that contain either DataContainers and/or
 58    AbstractDataContainers.
 59    """
 60
 61    def __init__(self,
 62                 metadata: PatkitBaseModel,
 63                 container: AbstractDataContainer | None = None,
 64                 file_info: FileInformation | None = None,
 65                 ) -> None:
 66        # The super().__init__() call below is needed to make sure that
 67        # inheriting classes which also inherit from UserDict and UserList are
 68        # initialised properly.
 69        super().__init__()
 70
 71        self._metadata = metadata
 72        self.container = container
 73        self._file_info = file_info
 74
 75    def __getstate__(self) -> dict:
 76        """
 77        Return this AbstractData's pickle compatible state.
 78
 79        To achieve pickle compatibility, subclasses should take care to delete
 80        any cyclical references, like is done with `self.container` here.
 81
 82        NOTE! This also requires container to be reset after unpickling by the
 83        owners, because the unpickled class can not know who owns it.
 84
 85        Returns
 86        -------
 87        dict
 88            The state without cyclical references.
 89        """
 90        state = self.__dict__.copy()
 91        del state['container']
 92        return state
 93
 94    @property
 95    @abc.abstractmethod
 96    def name(self) -> str:
 97        """
 98        Name of this instance.
 99
100        In most cases name is supposed to be implemented with the following
101        idiom:
102        ```python
103        return NAME_OF_THIS_CLASS.generate_name(self.metadata)
104        ```
105        For example, for PD:
106        ```python
107        return PD.generate_name(self.metadata)
108        ```
109
110        Returns
111        -------
112        str
113            Name of this instance.
114        """

384class AbstractDataContainer(AbstractDataObject):
385    """
386    Abstract baseclass for Recording, Session, and DataSet. 
387
388    This class collects behaviors that are shared by the data base classes i.e.
389    classes which collect DataContainers and/or AbstractDataContainers.
390    """
391
392    def __init__(self,
393                 name: str,
394                 metadata: PatkitBaseModel,
395                 container: AbstractDataContainer | None = None,
396                 file_info: FileInformation | None = None,
397                 statistics: dict[str, 'Statistic'] | None = None
398                 ) -> None:
399        super().__init__(
400            container=container, metadata=metadata, file_info=file_info)
401
402        self._name = name
403        self.statistics = {}
404        if statistics:
405            self.statistics.update(statistics)
406
407    @property
408    def name(self) -> str:
409        """
410        Name of this instance.
411
412        AbstractDataContainers get their names mainly from the file system. DataSets
413        are named after the root directory name, Sessions for the session
414        directories and Trials for the trial file names. 
415
416        Returns
417        -------
418        str
419            The name as string.
420        """
421        return self._name
422
423    def add_statistic(
424            self, statistic: 'Statistic', replace: bool = False) -> None:
425        """
426        Add a Statistic to this Session.
427
428        Parameters
429        ----------
430        statistic : Statistic
431            Statistic to be added.
432        replace : bool, optional
433            Should we replace any existing Statistic by the same name, by
434            default False
435
436        Raises
437        ------
438        OverwriteError
439            In case replace was False and there exists already a Statistic with
440            the same name in this Session.
441        """
442        name = statistic.name
443
444        if name in self.statistics and not replace:
445            raise OverwriteError(
446                "A Statistic named '" + name +
447                "' already exists and replace flag was False.")
448
449        if replace:
450            self.statistics[name] = statistic
451            _datastructures_logger.debug("Replaced statistic %s.", name)
452        else:
453            self.statistics[name] = statistic
454            _datastructures_logger.debug("Added new statistic %s.", name)
