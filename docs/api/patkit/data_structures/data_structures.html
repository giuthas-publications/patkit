  63class Answer(UserList):
  64    """
  65    Answer is an answer to an Exercise and consists of PatGrids.
  66
  67    The PatGrids correspond to the Recordings that the Exercise is based on.
  68    """
  69    def __init__(
  70        self,
  71        container: Exercise,
  72        scenario: Session,
  73        scramble: bool,
  74        cursor: int = 0,
  75    ):
  76        super().__init__()
  77        self.container = container
  78        self.scenario = scenario
  79        self.cursor = cursor
  80
  81        for recording in self.scenario:
  82            patgrid = deepcopy(recording.patgrid)
  83            self.append(patgrid)
  84
  85        if scramble:
  86            for patgrid in self:
  87                for tier in patgrid:
  88                    patgrid[tier].scramble()
  89            self.edited = [False for i in range(len(self))]
  90        else:
  91            self.edited = [True for i in range(len(self))]
  92
  93    def __repr__(self) -> str:
  94        representation = ""
  95        for patgrid in self:
  96            patgrid_string = f"{patgrid}"
  97            representation += indent(text=patgrid_string,prefix="\t")
  98        return representation
  99
 100    def current(self) -> PatGrid:
 101        """
 102        The PatGrid at the current index.
 103
 104        Returns
 105        -------
 106        PatGrid
 107            The PatGrid.
 108        """
 109        return self[self.cursor]
 110
 111    def go_to_recording(self, index: int) -> int:
 112        """
 113        Go to Recording at index.
 114
 115        Returns
 116        -------
 117        int
 118            index of current recording
 119        """
 120        if len(self) > index and index >= 0:
 121            self.cursor = index
 122        return self.cursor
 123
 124    def next(self) -> int:
 125        """
 126        Go to next Recording.
 127
 128        Returns
 129        -------
 130        int
 131            index of next recording
 132        """
 133        if len(self) > self.cursor + 1:
 134            self.cursor += 1
 135        return self.cursor
 136
 137    def previous(self) -> int:
 138        """
 139        Go to previous Recording.
 140
 141        Returns
 142        -------
 143        int
 144            index of previous recording
 145        """
 146        if self.cursor > 0:
 147            self.cursor -= 1
 148        return self.cursor

 151class Exercise(UserDict):
 152    """
 153    Exercise is list of Answers relating to a Scenario (Session).
 154
 155    If no example Answers are provided, the current PatGrids in the Session are
 156    copied and used as the example.
 157    """
 158
 159    def __init__(
 160        self,
 161        scenario: Session,
 162        answers: list[Answer] | None = None,
 163        example: dict[str, Answer] | None = None,
 164        index: int = 0,
 165    ):
 166        super().__init__()
 167        if example is None:
 168            _logger.debug("Creating an example answer")
 169            self.example = Answer(
 170                container=self,
 171                scenario=scenario,
 172                scramble = False,
 173            )
 174        else:
 175            self.example = example
 176        self.scenario = scenario
 177
 178        if answers is not None:
 179            self.update(answers)
 180
 181        self.cursor = index
 182
 183    def __repr__(self) -> str:
 184        representation = "Exercise:\n"
 185        scenario_repr = f"{self.scenario}"
 186        scenario_repr = indent(text=scenario_repr,prefix="\t\t")
 187        representation += f"\tScenario:\n{scenario_repr}\n"
 188        example_repr = f"{self.example}"
 189        example_repr = indent(text=example_repr,prefix="\t")
 190        representation += f"\tExample:\n{example_repr}\n"
 191        representation += "\tAnswers:\n"
 192        answers_string = ""
 193        for name in list(self.keys()):
 194            answers_string += f"{name}:\n{self[name]}"
 195        representation += indent(text=answers_string,prefix="\t\t")
 196        return representation
 197
 198    @property
 199    def current_answer(self) -> Answer:
 200        return self[list(self.keys())[self.cursor]]
 201
 202    def new_blank_answer(self, cursor: int = 0) -> None:
 203        """
 204        Create a new blank Answer and append it to this Exercise.
 205
 206        Parameters
 207        ----------
 208        cursor : int
 209            Cursor position for the new Answer.
 210        """
 211        _logger.debug("Creating a blank answer.")
 212        blank = Answer(
 213                container=self,
 214                scenario=self.scenario,
 215                scramble = True,
 216                cursor=cursor,
 217            )
 218        name = f"Answer {len(self)+1}"
 219        self[name] = blank

 222class Manifest(UserList):
 223    """
 224    Manifest is a list of Scenario paths as strings.
 225    """
 226    @staticmethod
 227    def read_manifest(path: Path) -> list[str] | None:
 228        """
 229        Static method for reading a Manifest files contents.
 230
 231        Parameters
 232        ----------
 233        path : Path
 234            Path of the file.
 235
 236        Returns
 237        -------
 238        list[str] | None
 239            Either the contents of the file or None if the file was empty.
 240        """
 241        if not path.exists():
 242            return None
 243        with open(path, 'r', encoding='UTF-8') as file:
 244            lines = [line.rstrip() for line in file]
 245            return lines
 246        return None
 247
 248    def __init__(self, path: Path | None = None):
 249        super().__init__()
 250        path = path.resolve()
 251        self.path = path
 252        if path is not None:
 253            scenario_paths = Manifest.read_manifest(path)
 254            if scenario_paths is not None:
 255                self.extend(scenario_paths)
 256
 257    @property
 258    def scenarios(self) -> list[str]:
 259        """
 260        The contents of this Manifest as a list.
 261
 262        Returns
 263        -------
 264        list[str]
 265            The contents.
 266        """
 267        return list(self)
 268
 269    def save(self):
 270        """
 271        Write this manifest into its file. 
 272        """
 273        if self.path is not None:
 274            with open(self.path, 'w', encoding='UTF-8') as file:
 275                for scenario_path in self:
 276                    file.write(f"{scenario_path}\n")

 281class Session(AbstractDataContainer, UserList):
 282    """
 283    The metadata and Recordings of a recording session.
 284
 285    This class behaves exactly like a list of Recordings with some extra
 286    fields. While some legacy code may be left behind, the preferred idiom for
 287    iterating over the recordings is `for recording in recording_session:`.
 288
 289    Sessions can also hold aggregate data in the form of Statistics.
 290    """
 291
 292    def __init__(
 293            self,
 294            name: str,
 295            config: SessionConfig,
 296            file_info: FileInformation,
 297            recordings: list[Recording] | None = None,
 298            statistics: dict[str, Statistic] | None = None
 299    ) -> None:
 300        super().__init__(
 301            container=None, name=name, metadata=config,
 302            file_info=file_info, statistics=statistics)
 303
 304        if not recordings is None:
 305            for recording in recordings:
 306                if not recording.container:
 307                    recording.container = self
 308            self.extend(recordings)
 309
 310        self.config = config
 311
 312    def __repr__(self) -> str:
 313        representation = "Session:\n"
 314        for recording in self:
 315            representation += f"\tRecording: {recording.basename},\n"
 316        return representation
 317
 318    @property
 319    def recordings(self) -> list[Recording]:
 320        """
 321        Property to access the list of Recordings directly.
 322
 323        Returns
 324        -------
 325        list[Recording]
 326            The list of this Session's Recordings.
 327        """
 328        return self.data

 331class Recording(AbstractDataContainer, UserDict):
 332    """
 333    A Recording is a dictionary of 0-n synchronised Modalities.
 334
 335    Adding modalities can be done by `recording[name] = modality`, but
 336    `recording.add_modality` is preferred as a safer way which checks for
 337    overwriting. The reason Recording is a dictionary is to make it possible to
 338    iterate with the idiom `for modality_name in recording`.
 339
 340    The recording also contains the non-modality-specific metadata
 341    (participant, speech content, etc.) as a dictionary, as well as the
 342    textgrid for the whole recording.
 343
 344    In general, inheriting should not be necessary, but if it is, inheriting
 345    classes should call `self._read_textgrid()` after calling
 346    `super.__init__()` (with correct arguments) and doing any updates to
 347    `self.meta['textgrid']` that are necessary.
 348    """
 349    container: Session
 350
 351    def __init__(
 352            self,
 353            metadata: RecordingMetaData,
 354            file_info: FileInformation,
 355            container: Session | None = None,
 356            excluded: bool = False,
 357    ) -> None:
 358        """
 359        Construct a mainly empty recording without modalities.
 360
 361        Modalities and annotations get added after constructions with their own
 362        add_[modality or annotation] functions.
 363
 364        NOTE: `after_modalities_init` should be called on each new Recording
 365        after modalities have been loaded. It ensures that there is at least a
 366        minimal TextGrid in place to facilitate GUI functions.
 367
 368        Parameters
 369        ----------
 370        metadata : RecordingMetaData
 371            Some of the contents of the meta data are available as properties.
 372        excluded : bool, optional
 373            _description_, by default False
 374        """
 375        if file_info.recorded_meta_file:
 376            name = file_info.recorded_meta_file
 377        else:
 378            name = file_info.patkit_meta_file
 379        super().__init__(
 380            container=container, name=name,
 381            metadata=metadata, file_info=file_info)
 382
 383        self.excluded = excluded
 384        self.textgrid_path = None
 385        self.textgrid = None
 386        self.patgrid = None
 387        self.annotations = {}
 388
 389    @property
 390    def modalities(self) -> dict[str, Modality]:
 391        """
 392        Dictionary of the modalities of this Recording.
 393
 394        Returns
 395        -------
 396        dict[str, Modality]
 397            The dictionary of the Modalities.
 398        """
 399        return self.data
 400
 401    @property
 402    def basename(self) -> str:
 403        """Filename of this Recording without extensions."""
 404        return self.file_info.basename
 405
 406    def _read_textgrid(self) -> textgrids.TextGrid | None:
 407        """
 408        Read the textgrid specified in self.meta.
 409
 410        If file does not exist or reading fails, recovery is attempted 
 411        by logging an error and creating an empty textgrid for this 
 412        recording.
 413        """
 414        textgrid = None
 415        if self.textgrid_path.is_file():
 416            try:
 417                textgrid = textgrids.TextGrid(self.textgrid_path)
 418                _logger.info("Read textgrid in %s.",
 419                             self.textgrid_path)
 420            except Exception as exception:
 421                _logger.critical(
 422                    "Could not read textgrid in %s.", self.textgrid_path)
 423                _logger.critical(
 424                    "Failed with: %s.", str(exception))
 425        else:
 426            notice = 'Note: ' + str(self.textgrid_path) + " did not exist."
 427            _logger.warning(notice)
 428        return textgrid
 429
 430    def identifier(self) -> str:
 431        """
 432        Generate a unique identifier for this Recording from metadata.
 433
 434        Returns
 435        -------
 436        str
 437            prompt followed by time of recording.
 438        """
 439        return (f"{self.metadata.prompt} "
 440                f"{str(self.metadata.time_of_recording)}")
 441
 442    def exclude(self) -> None:
 443        """
 444        Set `self.excluded` to True with a method.
 445
 446        This method exists to facilitate list comprehensions being used
 447        for excluding recordings e.g. 
 448        [recording.exclude() for recording in recordings if in some_list].
 449        """
 450        self.excluded = True
 451
 452    def write_textgrid(self, filepath: Path = None) -> None:
 453        """
 454        Save this recording's textgrid to file.
 455
 456        Keyword argument:
 457        filepath -- string specifying the path and name of the 
 458            file to be written. If filepath is not specified, this 
 459            method will try to overwrite the textgrid specified in 
 460            self.meta.
 461
 462            If filepath is specified, subsequent calls to this 
 463            function will write into the new path rather than 
 464            the original one.
 465        """
 466        try:
 467            if filepath:
 468                self.textgrid_path = filepath
 469                self.textgrid.write(filepath)
 470            else:
 471                self.textgrid.write(self.textgrid_path)
 472        except Exception as exception:
 473            _logger.critical(
 474                "Could not write textgrid to %s.", str(self.textgrid_path))
 475            _logger.critical(
 476                "TextGrid save failed with error: %s", str(exception))
 477
 478    def add_modality(
 479            self, modality: Modality, replace: bool = False
 480    ) -> None:
 481        """
 482        This method adds a new Modality object to the Recording.
 483
 484        Replacing a modality has to be specified otherwise if a
 485        Modality with the same name already exists in this Recording
 486        and the `replace` argument is not True, an Error is raised.
 487
 488        Arguments:
 489        modality -- object of type Modality to be added to 
 490            this Recording.
 491
 492        Keyword arguments:
 493        replace -- a boolean indicating if an existing Modality should
 494            be replaced.
 495        """
 496        name = modality.name
 497
 498        if name in self.modalities and not replace:
 499            raise OverwriteError(
 500                "A modality named " + name +
 501                " already exists and replace flag was False.")
 502
 503        if replace:
 504            self.modalities[name] = modality
 505            _logger.debug("Replaced modality %s.", name)
 506        else:
 507            self.modalities[name] = modality
 508            _logger.debug("Added new modality %s.", name)
 509
 510    def after_modalities_init(self) -> None:
 511        """
 512        Ensure everything is properly in place after loading modalities.
 513
 514        Currently, this is only used to create placeholder TextGrids when
 515        needed.
 516        """
 517        textgrid_path  = None
 518        if self.recorded_meta_path:
 519            textgrid_path = self.recorded_meta_path.with_suffix(
 520                SourceSuffix.TEXTGRID
 521            )
 522        elif self.recorded_data_path:
 523            textgrid_path = self.recorded_data_path.with_suffix(
 524                SourceSuffix.TEXTGRID
 525            )
 526
 527        if not textgrid_path or not textgrid_path.exists():
 528            if self.file_info.patkit_meta_file:
 529                textgrid_path = self.recorded_path
 530                name = Path(self.file_info.patkit_meta_file)
 531                textgrid_path = stem_path(textgrid_path/name)
 532                textgrid_path = textgrid_path.with_suffix(
 533                    SourceSuffix.TEXTGRID
 534                )
 535
 536        if textgrid_path is not None and textgrid_path.exists():
 537            self.textgrid_path = textgrid_path
 538            self.textgrid = self._read_textgrid()
 539
 540        if self.textgrid is None:
 541            if 'MonoAudio' in self.modalities:
 542                _logger.warning(
 543                    "%s: Creating a placeholder textgrid to make GUI function "
 544                    "correctly.", self.basename)
 545                textgrid = textgrids.TextGrid()
 546                interval = textgrids.Interval(
 547                    text=self.metadata.prompt,
 548                    xmin=self['MonoAudio'].min_time,
 549                    xmax=self['MonoAudio'].max_time,
 550                )
 551                tier = textgrids.Tier(
 552                    xmin=self['MonoAudio'].min_time,
 553                    xmax=self['MonoAudio'].max_time,
 554                )
 555                tier.append(interval)
 556                textgrid['Utterance'] = tier
 557                self.textgrid = textgrid
 558            else:
 559                _logger.warning("No audio found for %s.",
 560                                self.basename)
 561                _logger.warning("Can't create a textgrid so GUI may not "
 562                                "function correctly.")
 563
 564        if self.textgrid:
 565            self.patgrid = PatGrid(self.textgrid)
 566        else:
 567            self.patgrid = None
 568
 569
 570    def __str__(self) -> str:
 571        """
 572        Bare minimum string representation of the Recording.
 573
 574        Returns
 575        -------
 576        string
 577            'Recording [basename]'
 578        """
 579        return f"Recording {self.name}"
 580
 581    def __repr__(self) -> str:
 582        """Overrides the default implementation"""
 583        modalities = "{\n"
 584        for modality in self.modalities:
 585            modalities += ("\t" + modality
 586                           + f": {self.modalities[modality].name},\n")
 587        modalities += "}"
 588        return (
 589            f"Recording {self.basename}\n"
 590            f"{modalities}")
 591
 592    def __eq__(self, other):
 593        """Overrides the default implementation"""
 594        if isinstance(other, Recording):
 595            return self.name == other.name
 596        return NotImplemented

 599class Modality(AbstractData, OrderedDict):
 600    """
 601    Abstract superclass for all data Modality classes.
 602
 603    Any annotations associated with a Modality instance are accessible directly
 604    by `modality[annotation_type]`, because a Modality is also a OrderedDict of
 605    its Annotations.
 606    """
 607    container: Recording
 608
 609    @classmethod
 610    @abc.abstractmethod
 611    def generate_name(cls, params: ModalityMetaData) -> str:
 612        """Abstract version of generating a Modality name."""
 613
 614    def __init__(
 615            self,
 616            container: Recording,
 617            file_info: FileInformation,
 618            parsed_data: ModalityData | None = None,
 619            metadata: ModalityMetaData | None = None,
 620            time_offset: float | None = None,
 621            point_annotations: dict[AnnotationType, PointAnnotations] | None =
 622            None
 623    ) -> None:
 624        """
 625        Modality constructor.
 626
 627        Positional arguments:
 628        recording -- the containing Recording.
 629
 630        Keyword arguments:
 631        data_path -- path of the data file
 632        load_path -- path of data when saved by patkit - both data and metadata
 633        parent -- the Modality this one was derived from. None means this 
 634            is a recorded data Modality.
 635        parsed_data -- ModalityData object containing waveform, sampling rate,
 636            and either timevector and/or time_offset. 
 637        parsed_data -- a ModalityData object containing parsed data 
 638            that's been either read from file, loaded from file 
 639            (previously saved by patkit), or calculated from another modality.
 640            Providing a timevector 
 641            overrides any time_offset value given, but in absence of a 
 642            timevector the time_offset will be applied on reading the data 
 643            from file. 
 644        time_offset -- offset of this modality in relation to the Recordings
 645            baseline - usually the audio track. This will be ignored if 
 646            parsed_data exists and effectively overridden by 
 647            parsed_data.timevector.
 648        """
 649        super().__init__(
 650            container=container,
 651            metadata=metadata,
 652            file_info=file_info)
 653
 654        if point_annotations:
 655            self.update(point_annotations)
 656
 657        if parsed_data:
 658            self._modality_data = parsed_data
 659        else:
 660            self._modality_data = None
 661
 662        if (self._modality_data is None or
 663                self._modality_data.timevector is None):
 664            self._time_offset = time_offset
 665        else:
 666            self._time_offset = self._modality_data.timevector[0]
 667
 668        # This is a property which when set to True will also set
 669        # parent.excluded to True.
 670        self.excluded = False
 671        self._time_precision = None
 672
 673    @property
 674    def annotations(self) -> dict[AnnotationType, PointAnnotations]:
 675        """
 676        Property which is the annotations dictionary of this Modality.
 677
 678        This is currently the Modality itself.
 679
 680        Returns
 681        -------
 682        dict[AnnotationType, PointAnnotations]
 683            The dictionary.
 684        """
 685        return self
 686
 687    @property
 688    def recording(self) -> Recording | None:
 689        """
 690        This modality's container available with this alias for ease of use.
 691
 692        Returns
 693        -------
 694        Recording
 695            The Recording which contains this Modality.
 696        """
 697        return self.container
 698
 699    def _get_data(self) -> ModalityData:
 700        # TODO: Provide a way to force the data to be derived or loaded. this
 701        # would be used when parent modality has updated in some way. If this
 702        # is actually necessary to implement, a call back would be probably the
 703        # way to go.
 704        if self._file_info.recorded_data_file:
 705            return self._read_data()
 706
 707        if self._file_info.patkit_data_file:
 708            return self._load_data()
 709
 710        if self._metadata.parent_name:
 711            return self._derive_data()
 712
 713        raise MissingDataError(
 714            "Asked to get data but have no path and no parent Modality.\n"
 715            + "Don't know how to solve this.")
 716
 717    def _derive_data(self) -> ModalityData:
 718        """
 719        Derive data from another modality -- overridden by inheriting classes.
 720
 721        This method should be implemented by subclasses by calling 
 722        a suitable deriving function to provide on-the-fly loading of data.
 723
 724        This method is intended to rely on `self.parent` to provide the data
 725        to be processed.
 726        """
 727        raise NotImplementedError(
 728            "This method should be overridden by inheriting classes.")
 729
 730    def _load_data(self) -> ModalityData:
 731        """
 732        Load data from a saved file -- to be overridden by inheriting classes.
 733
 734        This method should be implemented by subclasses by calling 
 735        a suitable loading function to provide on-the-fly loading of data.
 736
 737        This method is intended to rely on self.load_path to know what to read.
 738        """
 739        raise NotImplementedError(
