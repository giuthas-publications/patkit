 48class SatAnnotation(ABC):

 59    @abstractmethod

 64        'Being at time' is defined in the sense of 'within epsilon of time'.

 69            The time in seconds to test against this Annotation.

127class SatInterval(SatAnnotation):
128    """TextGrid Interval representation to enable editing with GUI."""
129
130    @classmethod
131    def from_textgrid_interval(
132        cls,
133        interval: Interval,
134        prev_interval: None | Self,
135        next_interval: None | Self = None
136    ) -> Self:
137        """
138        Copy the info of a Python TextGrids Interval into a new SatInterval.
139
140        Only xmin and text are copied from the original Interval. xmax is
141        assumed to be handled by either the next SatInterval or the
142        constructing method if this is the last Interval. 
143
144        Since SatIntervals are doubly linked, an attempt will be made to link
145        prev and next to this interval. 
146
147        Returns the newly created SatInterval.
148        """
149        return cls(
150            begin=interval.xmin,
151            label=interval.text,
152            prev_interval=prev_interval,
153            next_interval=next_interval)
154
155    def __init__(self,
156                 begin: float,
157                 label: None | Transcript,
158                 prev_interval: None | Self = None,
159                 next_interval: None | Self = None) -> None:
160        super().__init__(
161            time=begin,
162            label=label,
163        )
164
165        self._prev_interval = prev_interval
166        if self.prev:
167            self.prev._next_interval = self
168
169        self._next_interval = next_interval
170        if self.next:
171            self.next.prev = self
172
173    def __repr__(self) -> str:
174        return (f"{self.__class__.__name__}: text: '{self.label}'\t "
175                f"begin: {self.begin}, end: {self.end}")
176
177    @property
178    def next(self) -> Self | None:
179        """The next annotation, if any."""
180        return self._next_interval
181
182    @next.setter
183    def next(self, next_interval: Self | None) -> None:
184        if next_interval != self._next_interval:
185            self._next_interval = next_interval
186
187    @property
188    def prev(self) -> Self | None:
189        """The previous annotation, if any."""
190        return self._prev_interval
191
192    @prev.setter
193    def prev(self, prev_interval: Self | None) -> None:
194        if prev_interval != self._prev_interval:
195            self._prev_interval = prev_interval
196
197    @property
198    def begin(self) -> float:
199        """Beginning time point of the interval."""
200        return self._time
201
202    @begin.setter
203    def begin(self, value: float) -> None:
204        self._time = value
205
206    @property
207    def mid(self) -> float | None:
208        """
209        Middle time point of the interval.
210
211        This is a property that will return None
212        if this Interval is the one that marks
213        the last boundary.
214        """
215        if self._next_interval:
216            return (self.begin + self._next_interval.begin)/2
217        return None
218
219    @property
220    def end(self) -> float | None:
221        """
222        End time point of the interval.
223
224        This is a property that will return None
225        if this Interval is the one that marks
226        the last boundary.
227        """
228        if self._next_interval:
229            return self._next_interval.begin
230        return None
231
232    @end.setter
233    def end(self, value: float) -> None:
234        if self._next_interval:
235            self._next_interval.begin = value
236
237    def is_at_time(self, time: float, epsilon) -> bool:
238        """
239        Intervals are considered equivalent if the difference between their
240        `begin` values is < epsilon. Epsilon is a constant defined in patkit's
241        configuration.
242        """
243        return abs(self.begin - time) < epsilon
244
245    def is_last(self) -> bool:
246        """Is this the last Interval in this Tier."""
247        return self._next_interval is None
248
249    def is_legal_value(self, time: float, epsilon: float) -> bool:
250        """
251        Check if the given time is between the previous and next boundary.
252
253        Usual caveats about float testing don't apply, because each boundary is
254        padded with patkit epsilon. Tests used do not include equality with
255        either bounding boundary, and that may or may not be trusted to be the
256        actual case depending on how small the epsilon is.
257
258        Returns True, if time is  between the previous and next boundary.
259        """
260        return (time + epsilon < self._next_interval.begin and
261                time > epsilon + self.prev.begin)
262
263    def contains(self, time: float) -> bool:
264        if self.begin < time < self.end:
265            return True
266        return False

269class SatTier(list):
270    """TextGrid Tier representation to enable editing with GUI."""
271
272    @classmethod
273    def from_textgrid_tier(cls, tier: Tier) -> Self:
274        """
275        Copy a Python TextGrids Tier as a SatTier.
276
277        Returns the newly created SatTier.
278        """
279        return cls(tier)
280
281    def __init__(self, tier: Tier) -> None:
282        super().__init__()
283        last_interval = None
284        prev = None
285        for interval in tier:
286            current = SatInterval.from_textgrid_interval(interval, prev)
287            self.append(current)
288            prev = current
289            last_interval = interval
290        self.append(SatInterval(last_interval.xmax, None, prev))
291
292    def __repr__(self) -> str:
293        representation = f"{self.__class__.__name__}:\n"
294        for interval in self:
295            representation += str(interval) + "\n"
296        return representation
297
298    @property
299    def begin(self) -> float:
300        """
301        Begin timestamp.
302
303        Corresponds to a TextGrid Interval's xmin.
304
305        This is a property and the actual value is generated from the first
306        SatInterval of this SatTier.
307        """
308        return self[0].begin
309
310    @property
311    def end(self) -> float:
312        """
313        End timestamp.
314
315        Corresponds to a TextGrid Interval's xmin.
316
317        This is a property and the actual value is generated from the last
318        SatInterval of this SatTier.
319        """
320        # This is slightly counterintuitive, but the last interval is in fact
321        # empty and only represents the final boundary. So its `begin` is the
322        # final boundary.
323        return self[-1].begin
324
325    @property
326    def is_point_tier(self) -> bool:
327        """Is this Tier a PointTier."""
328        return False
329
330    def boundary_at_time(
331            self, time: float, epsilon: float) -> SatInterval | None:
332        """
333        If there is a boundary at time, return it.
334
335        Returns None, if there is no boundary at time. 
336
337        'Being at time' is defined as being within patkit epsilon of the given
338        timestamp.
339        """
340        for interval in self:
341            if interval.is_at_time(time=time, epsilon=epsilon):
342                return interval
343        return None
344
345    def get_interval_by_category(
346        self,
347        interval_category: IntervalCategory,
348        label: str | None = None
349    ) -> SatInterval:
350        """
351        Return the Interval matching the category in this Tier.
352
353        If interval_category is FIRST_LABELED or LAST_LABELED, the label should
354        be specified as well.
355
356        Parameters
357        ----------
358        interval_category : IntervalCategory
359            The category to search for.
360        label : Optional[str], optional
361            Label to search for when doing a label based category search, by
362            default None
363
364        Returns
365        -------
366        SatInterval
367            _description_
368        """
369        if interval_category is IntervalCategory.FIRST_NON_EMPTY:
370            for interval in self:
371                if interval.label:
372                    return interval
373
374        if interval_category is IntervalCategory.LAST_NON_EMPTY:
375            for interval in reversed(self):
376                if interval.label:
377                    return interval
378
379        if interval_category is IntervalCategory.FIRST_LABELED:
380            for interval in self:
381                if interval.label == label:
382                    return interval
383
384        if interval_category is IntervalCategory.LAST_LABELED:
385            for interval in reversed(self):
386                if interval.label == label:
387                    return interval
388
389    def get_labels(self, time_vector: np.ndarray) -> np.ndarray:
390        """
391        Get the labels at the times in the `time_vector`.
392
393        Parameters
394        ----------
395        time_vector : np.ndarray
396            Time stamps to retrieve the labels for.
397
398        Returns
399        -------
400        np.ndarray
401            This array contains the labels as little endian Unicode strings.
402        """
403        max_label = max(
404            [len(element.label) for element in self
405             if element.label is not None]
406        )
407        labels = np.empty(len(time_vector), dtype=f"<U{max_label}")
408        for (i, time) in enumerate(time_vector):
409            labels[i] = self.label_at(time)
410        return labels
411
412    def label_at(self, time: float) -> str:
413        """
414        Get the label at the given time.
415
416        Parameters
417        ----------
418        time : float
419            Time in seconds to retrieve the label for.
420
421        Returns
422        -------
423        str
424            The label string.
425        """
426        if time < self.begin or time > self.end:
427            return ""
428
429        for element in self:
430            if element.contains(time):
431                return element.label

434class SatGrid(OrderedDict):
435    """
436    TextGrid representation which makes editing easier.
437
438    SatGrid is a OrderedDict very similar to Python textgrids TextGrid, but
439    made up of SatTiers that in turn contain intervals or points as doubly
440    linked lists instead of just lists. See the relevant classes for more
441    details.
442    """
443
444    def __init__(self, textgrid: TextGrid) -> None:
445        super().__init__()
446        for tier_name in textgrid:
447            self[tier_name] = SatTier.from_textgrid_tier(textgrid[tier_name])
448
449    # def as_textgrid(self):
450    #     pass
451
452    @property
453    def begin(self) -> float:
454        """
455        Begin timestamp.
456
457        Corresponds to a TextGrids xmin.
458
459        This is a property and the actual value is generated from the first
460        SatTier of this SatGrid.
461        """
462        key = list(self.keys())[0]
463        return self[key].begin
464
465    @property
466    def end(self) -> float:
467        """
468        End timestamp.
469
470        Corresponds to a TextGrids xmax.
471
472        This is a property and the actual value is generated from the first
473        SatTier of this SatGrid.
474        """
475        # First Tier
476        key = list(self.keys())[0]
477        # Return the end of the first Tier.
478        return self[key].end
479
480    def format_long(self) -> str:
481        """Format self as long format TextGrid."""
482        out = long_header.format(self.begin, self.end, len(self))
483        tier_count = 1
484        for name, tier in self.items():
485            if tier.is_point_tier:
486                tier_type = 'PointTier'
487                elem_type = 'points'
488            else:
489                tier_type = 'IntervalTier'
490                elem_type = 'intervals'
491            out += long_tier.format(tier_count,
492                                    tier_type,
493                                    name,
494                                    self.begin,
495                                    self.end,
496                                    elem_type,
497                                    len(tier)-1)
498            for elem_count, elem in enumerate(tier, 1):
499                if tier.is_point_tier:
500                    out += long_point.format(elem_count,
501                                             elem.time,
502                                             elem.label)
503                elif elem.next:
504                    out += long_interval.format(elem_count,
505                                                elem.begin,
506                                                elem.end,
507                                                elem.label)
508                else:
509                    # The last interval does not contain anything.
510                    # It only marks the end of the file and final
511                    # interval's end. That info got already used by
512                    # elem.end (which is elem.next.begin) above.
513                    pass
514        return out
515
516    def get_labels(self, time_vector: np.ndarray) -> dict[str, np.ndarray]:
517        """
518        Get the labels at the times in the `time_vector`.
519
520        Parameters
521        ----------
522        time_vector : np.ndarray
523            Time values to get the labels for.
524
525        Returns
526        -------
527        dict[str, np.ndarray]
528            Dictionary of the labels indexed by tier name.
529        """
530        labels = {}
531        for tier_name in self:
532            labels[tier_name] = self[tier_name].get_labels(time_vector)
533
534        return labels
