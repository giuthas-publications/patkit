 93    def __init__(
 94            self,
 95            configuration_paths: ConfigPaths
 96    ) -> None:
 97        """
 98        Init the main configuration object. 
 99
100        Run only once. Updates should be done with methods of the class.
101
102        Parameters
103        -------
104        configuration_paths : ConfigPaths
105            Paths to load the configuration from.
106        """
107        # TODO 0.24 do reporting and logging on what gets loaded and where
108        # from. this or similar for reporting
109        # https://stackoverflow.com/questions/24469662/how-to-redirect-logger-output-into-pyqt-text-widget
110        self._config_paths = configuration_paths
111
112        if self._config_paths.data_config is not None:
113            self._data_yaml = load_data_params(
114                self._config_paths.data_config)
115            self._data_config = DataConfig(**self._data_yaml.data)
116            recorded = self._config_paths.path/self._data_config.recorded_data_path
117            self._data_config.recorded_data_path = recorded.resolve()
118        else:
119            self._data_config = None
120
121        if self._config_paths.gui_config is not None:
122            self._gui_yaml = load_gui_params(self._config_paths.gui_config)
123            self._gui_config = GuiConfig(**self._gui_yaml.data)
124        else:
125            self._gui_config = None
126
127        if self._config_paths.publish_config is not None:
128            self._publish_yaml = load_publish_params(
129                self._config_paths.publish_config)
130            self._publish_config = PublishConfig(**self._publish_yaml.data)
131        else:
132            self._publish_config = None
133
134        if self._config_paths.simulation_config is not None:
135            self._simulation_yaml = load_simulation_params(
136                self._config_paths.simulation_config)
137            self._simulation_config = SimulationConfig(
138                **self._simulation_yaml.data)
139        else:
140            self._simulation_config = None
141
142        # self._plot_yaml = load_plot_params(config['plotting_parameter_file'])
143        # self._plot_config = PlotConfig(**self._plot_yaml.data)

275        Update the configuration from a file.
276
277        This first updates the main configuration and then recursively updates

279
280        NOTE: comment round tripping may/will be broken by running any of the

282

285        configuration_paths : ConfigPaths

288        self._config_paths = configuration_paths

290            self.update_data_config_from_file(
291                self.config_paths.data_config
292            )
293        if self.config_paths.gui_config is not None:
294            self.update_gui_from_file(self._config_paths.gui_config)
295        if self.config_paths.publish_config is not None:
296            self.update_publish_from_file(
297                self._config_paths.publish_config
298            )
