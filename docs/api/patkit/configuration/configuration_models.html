These are the code friendly wrappers for the configuration structures.

configuration_parsers contains the actual parsing of strictyaml into comment-retaining dictionary-like structures. Here those structures get parsed into pydantic models that know what their fields actually are.

This two level structure is maintained so that in some version after patkit 1.0 we can implement configuration round tripping with preserved comments.

145class SearchPattern(UpdatableBaseModel):
146    """
147    Representation for simple and regexp search patterns.
148
149    Members
150    ----------
151    pattern : str
152        The pattern to search for
153    is_regexp : bool, optional
154        If the pattern should be treated as a regexp or not. Defaults to False.
155    """
156    pattern: str
157    is_regexp: bool = False
158
159    def match(self, string: str) -> bool:
160        """
161        Match this pattern to the argument string.
162
163        If this pattern is not a regexp then this method will return True only
164        when the pattern is verbatim the argument string.
165
166        Parameters
167        ----------
168        string : str
169            The string to match to.
170
171        Returns
172        -------
173        bool
174            True if this pattern matches the argument.
175        """
176        if self.is_regexp:
177            return bool(re.match(self.pattern, string))
178
179        return self.pattern == string
180
181    def search(self, string: str) -> bool:
182        """
183        Search for this pattern in the argument string.
184
185        If this pattern is not a regexp then this method will return True only
186        when the pattern is found verbatim in the argument string.
187
188        Parameters
189        ----------
190        string : str
191            The string to match to.
192
193        Returns
194        -------
195        bool
196            True if this pattern is found in the argument.
197        """
198        if self.is_regexp:
199            return bool(re.search(self.pattern, string))
200
201        return self.pattern in string
202
203    @staticmethod
204    def build(value: dict | str) -> 'SearchPattern':
205        """
206        Build a SearchPattern from a dictionary or a string.
207
208        Parameters
209        ----------
210        value : dict | str
211            The dictionary or string to build the SearchPattern from. If the
212            parameter is a string it is used as the pattern, if it is a dict
213            it's passed as keyword arguments to the SearchPattern constructor.
214
215        Returns
216        -------
217        SearchPattern
218            The constructed SearchPattern.
219
220        Raises
221        ------
222        ValueError
223            If the value is not a dict or a string.
224        """
225        if isinstance(value, str):
226            return SearchPattern(pattern=value)
227
228        if isinstance(value, dict):
229            return SearchPattern(**value)
230
231        raise ValueError("Unrecognised input value type: "
232                         + type(value).__name__ + ".")

235class TimeseriesNormalisation(UpdatableBaseModel):
236    """
237    Selection between peak normalised, bottom normalised or both.
238
239    Contains a boolean for each peak and bottom normalisation.
240    """
241    peak: bool = False
242    bottom: bool = False
243
244    # TODO: this class needs a special dumper to save things correctly.
245
246    @staticmethod
247    def build(
248            value: str
249    ) -> 'TimeseriesNormalisation':
250        """
251        Construct a TimeseriesNormalisation object from a string value.
252
253        The value usually comes from a config file.
254
255        Parameters
256        ----------
257        value : str
258            'none': no normalisation
259            'peak': divide all data points y-values by the largest y-value
260            'bottom': deduct the lowest y-value from all data points y-values
261            'both': do first bottom normalisation and then peak normalisation.
262
263        Returns
264        -------
265        TimeseriesNormalisation
266            The new TimeseriesNormalisation with fields set as expected.
267        """
268        match value:
269            case 'none':
270                return TimeseriesNormalisation()
271            case 'peak':
272                return TimeseriesNormalisation(peak=True)
273            case 'bottom':
274                return TimeseriesNormalisation(bottom=True)
275            case 'both':
276                return TimeseriesNormalisation(peak=True, bottom=True)
277            case _:
278                raise ValueError("Unrecognised value: " + value + ".")
