148class SearchPattern(UpdatableBaseModel):
149    """
150    Representation for simple and regexp search patterns.
151
152    Members
153    ----------
154    pattern : str
155        The pattern to search for
156    is_regexp : bool, optional
157        If the pattern should be treated as a regexp or not. Defaults to False.
158    """
159    pattern: str
160    is_regexp: bool = False
161
162    def match(self, string: str) -> bool:
163        """
164        Match this pattern to the argument string.
165
166        If this pattern is not a regexp then this method will return True only
167        when the pattern is verbatim the argument string.
168
169        Parameters
170        ----------
171        string : str
172            The string to match to.
173
174        Returns
175        -------
176        bool
177            True if this pattern matches the argument.
178        """
179        if self.is_regexp:
180            return bool(re.match(self.pattern, string))
181
182        return self.pattern == string
183
184    def search(self, string: str) -> bool:
185        """
186        Search for this pattern in the argument string.
187
188        If this pattern is not a regexp then this method will return True only
189        when the pattern is found verbatim in the argument string.
190
191        Parameters
192        ----------
193        string : str
194            The string to match to.
195
196        Returns
197        -------
198        bool
199            True if this pattern is found in the argument.
200        """
201        if self.is_regexp:
202            return bool(re.search(self.pattern, string))
203
204        return self.pattern in string
205
206    @staticmethod
207    def build(value: dict | str) -> 'SearchPattern':
208        """
209        Build a SearchPattern from a dictionary or a string.
210
211        Parameters
212        ----------
213        value : dict | str
214            The dictionary or string to build the SearchPattern from. If the
215            parameter is a string it is used as the pattern, if it is a dict
216            it's passed as keyword arguments to the SearchPattern constructor.
217
218        Returns
219        -------
220        SearchPattern
221            The constructed SearchPattern.
222
223        Raises
224        ------
225        ValueError
226            If the value is not a dict or a string.
227        """
228        if isinstance(value, str):
229            return SearchPattern(pattern=value)
230
231        if isinstance(value, dict):
232            return SearchPattern(**value)
233
234        raise ValueError("Unrecognised input value type: "
235                         + type(value).__name__ + ".")
